    \hfill(1980)
    \begin{enumerate}
        \item Reflection about the line $y=x$.
        \item Translation through a distances of x-axis.
        \item Rotation through an $\frac{\pi}{4}$ about the origin in the counter clockwise direction.
    \end{enumerate}
